import java.util.Scanner;

class StudentGradeCalculator {

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);

        double subject1Marks;
        double subject2Marks;
        double subject3Marks;
        double totalMarks;
        double percentage;
        String grade;

        System.out.print("Enter marks of Subject 1: ");
        subject1Marks = input.nextDouble();

        System.out.print("Enter marks of Subject 2: ");
        subject2Marks = input.nextDouble();

        System.out.print("Enter marks of Subject 3: ");
        subject3Marks = input.nextDouble();

        totalMarks = subject1Marks + subject2Marks + subject3Marks;

        double totalSubjects = 3;
        percentage = totalMarks / totalSubjects;

        if (percentage >= 90) {
            grade = "A";
        } else if (percentage >= 75) {
            grade = "B";
        } else if (percentage >= 50) {
            grade = "C";
        } else {
            grade = "Fail";
        }

        System.out.println("Total Marks: " + totalMarks);
        System.out.println("Percentage: " + percentage);
        System.out.println("Grade: " + grade);
    }
}
